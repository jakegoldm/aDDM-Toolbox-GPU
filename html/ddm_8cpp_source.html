315DDMTrial DDM::simulateTrial(int ValueLeft, int ValueRight, int timeStep) {
316 float RDV = this->bias;
317 int time = 0;
318 int elapsedNDT = 0;
319 int RT;
320 int choice;
321 float mean;
322 std::vector<float>RDVs = {RDV};
323 std::random_device rd;
324 std::mt19937 gen(rd());
325
326 while (true) {
327 if (RDV >= this->barrier || RDV <= -this->barrier) {
328 RT = time * timeStep;
329 if (RDV >= this->barrier) {
330 choice = -1;
331 } else {
332 choice = 1;
333 }
334 break;
335 }
336 if (elapsedNDT < this->nonDecisionTime / timeStep) {
337 mean = 0;
338 elapsedNDT += 1;
339 }
340 else {
341 mean = this->d * (ValueLeft - ValueRight);
342 }
343 std::normal_distribution<float> dist(mean, this->sigma);
344 float inc = dist(gen);
345 RDV += inc;
346 RDVs.push_back(RDV);
347 time += 1;
348 }
349 DDMTrial trial = DDMTrial(RT, choice, ValueLeft, ValueRight);
350 trial.RDVs = RDVs;
351 trial.timeStep = timeStep;
352 return trial;
353}
